function doGet(e) {
  var type = e.parameter.type || "price";
  var sym  = e.parameter.s    || "";

  // Decode %5E back to ^ for Yahoo Finance (indices: %5ENSEI, %5EBSESN, %5ENSEBANK)
  sym = sym.replace(/%5E/gi, "^");

  try {
    if (type === "batch")   return batchFetch(sym);
    if (type === "news")    return fetchNews(sym);
    if (type === "price")   return singleFetch(sym);
    if (type === "history") return historyFetch(sym);
  } catch(err) {
    return json({error: err.toString()});
  }
}

// ── SINGLE PRICE ──
function singleFetch(sym) {
  var url  = "https://query1.finance.yahoo.com/v8/finance/chart/" + encodeURIComponent(sym);
  var resp = UrlFetchApp.fetch(url, {muteHttpExceptions: true});
  return ContentService
    .createTextOutput(resp.getContentText())
    .setMimeType(ContentService.MimeType.JSON);
}

// ── BATCH PRICE (parallel) ──
function batchFetch(symbols) {
  var syms = symbols.split(",").map(function(s){ return s.trim(); });

  var requests = syms.map(function(s) {
    return {
      url: "https://query1.finance.yahoo.com/v8/finance/chart/" + encodeURIComponent(s),
      muteHttpExceptions: true
    };
  });

  var responses = UrlFetchApp.fetchAll(requests);
  var result    = {};

  syms.forEach(function(sym, i) {
    try {
      var j = JSON.parse(responses[i].getContentText());
      if (j.chart && j.chart.result) {
        var m = j.chart.result[0].meta;
        result[sym] = {
          price:      m.regularMarketPrice,
          prevClose:  m.chartPreviousClose,
          open:       m.regularMarketOpen,
          high:       m.regularMarketDayHigh,
          low:        m.regularMarketDayLow,
          week52High: m.fiftyTwoWeekHigh,
          week52Low:  m.fiftyTwoWeekLow,
          volume:     m.regularMarketVolume,
          pe:         m.trailingPE,
          eps:        m.epsTrailingTwelveMonths,
          mktCap:     m.marketCap
        };
      } else {
        result[sym] = null;
      }
    } catch(e) {
      result[sym] = null;
    }
  });

  return json(result);
}

// ── NEWS (Google RSS) ──
function fetchNews(sym) {
  var cleanSym = sym.replace(".NS","").replace(".BO","");
  var url = "https://news.google.com/rss/search?q="
          + encodeURIComponent(cleanSym + " NSE stock India")
          + "&hl=en-IN&gl=IN&ceid=IN:en";

  var resp = UrlFetchApp.fetch(url, {muteHttpExceptions: true});
  var xml  = resp.getContentText();

  var items   = [];
  var pattern = /<item>([\s\S]*?)<\/item>/g;
  var match;

  while ((match = pattern.exec(xml)) !== null && items.length < 5) {
    var item    = match[1];
    var title   = (item.match(/<title><!\[CDATA\[(.*?)\]\]><\/title>/)   || [])[1] || "";
    var link    = (item.match(/<link>(.*?)<\/link>/)                      || [])[1] || "";
    var pubDate = (item.match(/<pubDate>(.*?)<\/pubDate>/)                || [])[1] || "";
    if (title) items.push({title: title, link: link, date: pubDate});
  }

  return json({sym: sym, news: items});
}

// ── HISTORY FETCH (OHLCV for RSI/MACD) ──
function historyFetch(sym) {
  var range    = "30d";
  var interval = "1d";
  var url = "https://query1.finance.yahoo.com/v8/finance/chart/"
          + encodeURIComponent(sym)
          + "?range=" + range
          + "&interval=" + interval;

  try {
    var resp = UrlFetchApp.fetch(url, {muteHttpExceptions: true});
    var data = JSON.parse(resp.getContentText());

    if (!data.chart || !data.chart.result || !data.chart.result[0]) {
      return json({error: "No data"});
    }

    var result     = data.chart.result[0];
    var timestamps = result.timestamp;
    var quote      = result.indicators.quote[0];

    return json({
      dates:  timestamps,
      open:   quote.open,
      high:   quote.high,
      low:    quote.low,
      close:  quote.close,
      volume: quote.volume
    });

  } catch(err) {
    return json({error: err.toString()});
  }
}

// ── HELPER ──
function json(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
